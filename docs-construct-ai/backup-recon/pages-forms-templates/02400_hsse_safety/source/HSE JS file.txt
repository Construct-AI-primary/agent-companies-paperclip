// Hook for HSSE Form Management
const useHSSEForm = (contractorName, onFormSubmit) => {
const [generalInfo, setGeneralInfo] = useState({
supplierName: contractorName || '',
evaluationDate: new Date().toISOString().split('T')[0],
evaluator: '',
evaluatorCompany: ''
});

const [sectionsData, setSectionsData] = useState({
1: { answers: {}, completed: 0, score: 0, weight: 10 },
2: { answers: {}, completed: 0, score: 0, weight: 15 },
3: { answers: {}, completed: 0, score: 0, weight: 15 },
4: { answers: {}, completed: 0, score: 0, weight: 15 },
5: { answers: {}, completed: 0, score: 0, weight: 10 },
6: { answers: {}, completed: 0, score: 0, weight: 10 },
7: { answers: {}, completed: 0, score: 0, weight: 8 },
8: { answers: {}, completed: 0, score: 0, weight: 5 },
9: { answers: {}, completed: 0, score: 0, weight: 4 }
});

const [overallScore, setOverallScore] = useState(0);
const [maxScore, setMaxScore] = useState(0);
const [percentage, setPercentage] = useState(0);
const [recommendation, setRecommendation] = useState('No Score Yet');

const sectionQuestions = {
1: 3, 2: 10, 3: 19, 4: 23, 5: 7, 6: 17, 7: 9, 8: 4, 9: 1
};

// Calculate max score and overall score
useEffect(() => {
const totalMaxScore = Object.entries(sectionQuestions).reduce((total, [section, questions]) => {
return total + (questions * 5 * (sectionsData[section].weight / 100));
}, 0);

setMaxScore(totalMaxScore);

const totalScore = Object.entries(sectionsData).reduce((total, [sectionId, section]) => {
const weightedScore = section.score * (section.weight / 100);
return total + weightedScore;
}, 0);

setOverallScore(totalScore);

const percentageValue = totalMaxScore > 0 ? (totalScore / totalMaxScore) * 100 : 0;
setPercentage(percentageValue);

// Determine recommendation
if (percentageValue >= 90) {
setRecommendation('APPROVED - Excellent Performance');
} else if (percentageValue >= 80) {
setRecommendation('APPROVED - Good Performance');
} else if (percentageValue >= 70) {
setRecommendation('CONDITIONAL APPROVAL - Acceptable with Improvements');
} else if (percentageValue >= 60) {
setRecommendation('CONDITIONAL APPROVAL - Needs Significant Improvements');
} else {
setRecommendation('REJECTED - Does Not Meet Minimum Requirements');
}
}, [sectionsData]);

const updateSectionAnswer = (sectionId, questionId, field, value) => {
setSectionsData(prev => {
// Create deep clones to ensure React detects changes
const newSectionsData = { ...prev };
const section = newSectionsData[sectionId] || {
answers: {}, completed: 0, score: 0, weight: sectionsData[sectionId]?.weight || 10
};

if (!section.answers) section.answers = {};
if (!section.answers[questionId]) section.answers[questionId] = {};

// Set the field value
section.answers[questionId][field] = value;

newSectionsData[sectionId] = section;

// Recalculate completed questions and section score
const questionsInSection = sectionQuestions[sectionId] || 0;
let completed = 0;
let totalScore = 0;

Object.values(section.answers).forEach(answer => {
if (answer.answer && answer.answer.trim()) {
completed++;
if (typeof answer.score === 'number') {
totalScore += answer.score;
}
}
});

newSectionsData[sectionId].completed = completed;
newSectionsData[sectionId].score = totalScore;

return newSectionsData;
});

// Real-time save to database (only if not in auto-fill mode)
// We'll handle saving for auto-fill separately
};

const saveAnswerToDB = async (sectionId, questionId, field, value) => {
try {
const payload = {
contractor_name: contractorName,
document_name: ${contractorName} - Manual Entry Q${questionId},
document_type: HSSE Section ${sectionId} Question ${questionId},
document_path: /manual/questions/${sectionId}/${questionId},
processing_status: 'Manual Entry',
score: sectionsData[sectionId].answers[questionId]?.score || Math.round(typeof value === 'number' ? value : 0),
confidence: 100,
commentary: Manual entry for HSSE Question ${questionId} (Section ${sectionId}),
form_submission_data: {
question_id: questionId,
question_text: sectionsData[sectionId].answers[questionId]?.answer || '',
section: sectionId,
answer: field === 'answer' ? value : (sectionsData[sectionId].answers[questionId]?.answer || ''),
score: field === 'score' ? Math.round(value) : (sectionsData[sectionId].answers[questionId]?.score || 0),
feedback: field === 'feedback' ? value : (sectionsData[sectionId].answers[questionId]?.feedback || ''),
field_type: field,
manual_entry: true,
timestamp: new Date().toISOString()
}
};

const response = await fetch('/api/auto-fill/save-individual-answer', {
method: 'POST',
headers: {
'Content-Type': 'application/json'
},
body: JSON.stringify(payload)
});

if (!response.ok) {
console.error(Failed to save ${field} for question ${questionId});
} else {
console.log(✅ Saved ${field} for question ${questionId});
}
} catch (error) {
console.error(Error saving ${field} for question ${questionId}:, error);
}
};

const submitForm = async () => {
const formData = {
overallScore: overallScore,
percentage: percentage,
recommendation: recommendation,
sections: sectionsData,
generalInfo: generalInfo,
questions: Object.entries(sectionsData).reduce((all, [sectionId, section]) => {
Object.entries(section.answers).forEach(([questionId, data]) => {
all[questionId] = {
...data,
section: sectionId
};
});
return all;
}, {}),
formData: {}, // Legacy compatibility
selectedContractorId: contractorName,
currentSection: 1,
contractorName: contractorName,
timestamp: new Date().toISOString()
};

if (onFormSubmit) {
onFormSubmit(formData);
}
};

return {
generalInfo,
setGeneralInfo,
sectionsData,
overallScore,
maxScore,
percentage,
recommendation,
updateSectionAnswer,
submitForm,
sectionQuestions
};
};

// Auto Fill Documents Modal Component
const AutoFillModal = ({ isOpen, onClose, onFilesProcessed, contractorName }) => {
const [uploadedFiles, setUploadedFiles] = useState([]);
const [isProcessing, setIsProcessing] = useState(false);
const [additionalContext, setAdditionalContext] = useState('');
const [referenceUrls, setReferenceUrls] = useState(['']);
const fileInputRef = useRef(null);

const handleFileSelect = (files) => {
const validFiles = files.filter(file =>
['application/pdf', 'application/vnd.openxmlformats-officedocument.wordprocessingml.document', 'text/plain'].includes(file.type) ||
['.pdf', '.docx', '.doc', '.txt'].some(ext => file.name.toLowerCase().endsWith(ext))
);

setUploadedFiles(prev => [...prev, ...validFiles]);

if (validFiles.length < files.length) {
alert('Some files were skipped. Only PDF, DOCX, and TXT files are supported for auto-fill.');
}
};

const removeFile = (index) => {
setUploadedFiles(prev => prev.filter((_, i) => i !== index));
};

const addReferenceUrl = () => {
setReferenceUrls(prev => [...prev, '']);
};

const updateReferenceUrl = (index, value) => {
setReferenceUrls(prev => prev.map((url, i) => i === index ? value : url));
};

const removeReferenceUrl = (index) => {
setReferenceUrls(prev => prev.filter((_, i) => i !== index));
};

const processDocuments = async () => {
if (uploadedFiles.length === 0) {
alert('Please upload at least one document.');
return;
}

setIsProcessing(true);

try {
const formData = new FormData();
uploadedFiles.forEach(file => formData.append('documents', file));
formData.append('contractorName', contractorName || '');
formData.append('additionalContext', additionalContext);
formData.append('referenceUrls', JSON.stringify(referenceUrls.filter(url => url.trim())));

const response = await fetch('/api/auto-fill/process-documents', {
method: 'POST',
body: formData
});

if (!response.ok) {
throw new Error(HTTP ${response.status}: ${response.statusText});
}

const result = await response.json();

if (result.success && result.answers) {
onFilesProcessed(result.answers);
onClose();
alert(Auto-fill completed! ${result.answers.length} answers have been populated from the document(s).);
} else {
throw new Error(result.error || 'Failed to process documents');
}

} catch (error) {
console.error('Auto-fill error:', error);
alert(Auto-fill failed: ${error.message});
} finally {
setIsProcessing(false);
}
};

if (!isOpen) return null;

return (
<ModalOverlay>
<ModalContent>
<ModalHeader>
<h3 style={{ margin: 0 }}>Auto Fill from Documents</h3>
<CloseButton onClick={onClose}>×</CloseButton>
</ModalHeader>

<ModalBody>
<Description>
Upload supplier documents (PDF, DOCX, TXT) to automatically populate form answers using AI analysis.
</Description>

{/* Additional Context */}
<ModalFormGroup>
<Label>Additional Context (Optional):</Label>
<TextArea
rows={3}
value={additionalContext}
onChange={(e) => setAdditionalContext(e.target.value)}
placeholder="Enter any additional context or notes about the documents..."
/>
</ModalFormGroup>

{/* Reference URLs */}
<ModalFormGroup>
<Label>Reference URLs (Optional):</Label>
{referenceUrls.map((url, index) => (
<UrlInputContainer key={index}>
<UrlInput
type="url"
value={url}
onChange={(e) => updateReferenceUrl(index, e.target.value)}
placeholder="https://example.com/document"
/>
{referenceUrls.length > 1 && (
<RemoveButton onClick={() => removeReferenceUrl(index)}>×</RemoveButton>
)}
</UrlInputContainer>
))}
<AddButton onClick={addReferenceUrl}>+ Add Another URL</AddButton>
</ModalFormGroup>

{/* File Upload Area */}
<FileUploadArea onClick={() => fileInputRef.current?.click()}>
<UploadIcon>📄</UploadIcon>
<UploadText>Drop files here or click to upload</UploadText>
<SupportedText>Supported: PDF, DOCX, TXT</SupportedText>
<HiddenFileInput
ref={fileInputRef}
type="file"
multiple
accept=".pdf,.docx,.doc,.txt"
onChange={(e) => handleFileSelect(Array.from(e.target.files))}
/>
</FileUploadArea>

{/* Uploaded Files List */}
{uploadedFiles.length > 0 && (
<FilesList>
<h4>Uploaded Files:</h4>
{uploadedFiles.map((file, index) => (
<FileCard key={index}>
